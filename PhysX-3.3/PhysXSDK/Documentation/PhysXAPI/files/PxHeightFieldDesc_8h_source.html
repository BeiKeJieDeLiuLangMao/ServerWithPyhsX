169 PX_INLINE bool PxHeightFieldDesc::isValid() const
170 {
171  if (nbColumns < 2)
172  return false;
173  if (nbRows < 2)
174  return false;
175  switch (format)
176  {
177  case PxHeightFieldFormat::eS16_TM:
178  if (samples.stride < 4)
179  return false;
180  break;
181  default:
182  return false;
183  }
184  if (convexEdgeThreshold < 0)
185  return false;
186  if ((flags & PxHeightFieldFlag::eNO_BOUNDARY_EDGES) != flags)
187  return false;
188  if (thickness < -PX_MAX_BOUNDS_EXTENTS || thickness > PX_MAX_BOUNDS_EXTENTS)
189  return false;
190  return true;
191 }
